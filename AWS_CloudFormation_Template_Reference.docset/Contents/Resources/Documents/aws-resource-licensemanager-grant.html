AWS::LicenseManager::Grant

Specifies a grant.

A grant shares the use of license entitlements with specific AWS accounts. For more information, see Granted licenses in the AWS License Manager User Guide.

Syntax

To declare this entity in your AWS CloudFormation template, use the following syntax:

JSON

{
  "Type" : "AWS::LicenseManager::Grant",
  "Properties" : {
      "AllowedOperations" : AllowedOperationList,
      "ClientToken" : String,
      "Filters" : FilterList,
      "GrantArns" : ArnList,
      "GrantedOperations" : AllowedOperationList,
      "GranteePrincipalArn" : String,
      "GrantName" : String,
      "GrantStatus" : String,
      "HomeRegion" : String,
      "LicenseArn" : String,
      "MaxResults" : Integer,
      "NextToken" : String,
      "ParentArn" : String,
      "Principals" : ArnList,
      "SourceVersion" : String,
      "Status" : String,
      "StatusReason" : String,
      "Tags" : TagList,
      "Version" : String
    }
}
